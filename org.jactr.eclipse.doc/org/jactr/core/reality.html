/**
 this must be called before runtime.getController().run() is called. However,
 it does not matter if models are added before or after this.
*/
 public void setupReality()
 {
  ACTRRuntime runtime = ACTRRuntime.getDefaultRuntime();
  RealityInterface realInterface = runtime.getRealityInterface();
  
  //SwingRealityInterfaceComponent provides the eyes and hands for
  // interacting with Swing GUIs
  
  SwingRealityInterfaceComponent swingRIC = new SwingRealityInterfaceComponent(realInterface);
  
  //we can add multiple reality interfaces.. but you'd better know what
  // you're doing..
   
  realInterface.addRealityInterfaceComponent(swingRIC);
  
  //the configuration method will be called by the runtime just before
  // the models are executed
 }
